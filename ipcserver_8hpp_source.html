202 struct RequestHandler<Func> {
203  static ResultCode Handle(T *object, IPC::Message msg) {
204  TransactionFormat fmt;
205  std::tuple<AccessorHelper<Args>...> accessors = FormatBuilder<ArgPack<Args...>>::Build(fmt);
206 
207  fmt.rq.raw_data = new uint8_t[fmt.rq.raw_data_size];
208  fmt.rq.num_buffers = fmt.buffers.size();
209  fmt.rq.buffers = fmt.buffers.data();
210 
211  fmt.rs.raw_data = new uint8_t[fmt.rs.raw_data_size];
212  fmt.rs.objects = new ipc_server_object_t*[fmt.rs.num_objects];
213  fmt.out_objects = new Object*[fmt.rs.num_objects];
214 
215  ResultCode r = ipc_unflatten_request(&msg.msg, &fmt.rq, &object->object);
216  if(!r.IsOk()) {
217  // this gets forwarded to dispatch_shim, which will close the session for us
218  return r;
219  }
220 
221  r = RequestHandler<Func>::Helper(object, fmt, accessors, std::index_sequence_for<Args...>());
222  if(!r.IsOk()) {
223  ipc_response_t rs = ipc_default_response;
224  rs.result_code = r.code;
225  return ipc_server_object_reply(&object->object, &rs);
226  }
227 
228  for(size_t i = 0; i < fmt.rs.num_objects; i++) {
229  fmt.rs.objects[i] = &fmt.out_objects[i]->object;
230  }
231 
232  return ipc_server_object_reply(&object->object, &fmt.rs);
233  }
234  private:

236  static ResultCode Helper(T *object, TransactionFormat &fmt, const std::tuple<AccessorHelper<Args>...> &accessors, std::index_sequence<I...>) {
